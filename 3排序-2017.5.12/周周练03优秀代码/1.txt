#include<iostream>
#include<string>
#include <algorithm>
using namespace std;
class p{
public:
string color;
int number;
};
bool cmp(p a,p b)
{
	return a.number>b.number;
}
int main()
{
int T,n,i;
p pp[11];
cin>>T;
while(T--)
{
cin>>n;
for(i=0;i<n;i++)
cin>>pp[i].color>>pp[i].number;

sort(pp,pp+n,cmp);

for(i=0;i<n-1;i++)
cout<<pp[i].color<<" ";
cout<<pp[n-1].color<<endl;
}
return 0;
}
